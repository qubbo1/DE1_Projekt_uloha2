library ieee;
use ieee.std_logic_1164.all;

entity tb_debounce is
end tb_debounce;

architecture tb of tb_debounce is

    component debounce
        port (
            clk         : in  std_logic;
            rst         : in  std_logic;
            btn_in      : in  std_logic;
            btn_state   : out std_logic;
            btn_press   : out std_logic;
            btn_release : out std_logic
        );
    end component;

    signal clk         : std_logic := '0';
    signal rst         : std_logic := '1';
    signal btn_in      : std_logic := '0';
    signal btn_state   : std_logic;
    signal btn_press   : std_logic;
    signal btn_release : std_logic;

    constant CLK_T : time := 10 ns;  -- 100 MHz clock

begin

    --------------------------------------------------------------------
    -- DUT
    --------------------------------------------------------------------
    dut : debounce
        port map (
            clk         => clk,
            rst         => rst,
            btn_in      => btn_in,
            btn_state   => btn_state,
            btn_press   => btn_press,
            btn_release => btn_release
        );

    --------------------------------------------------------------------
    -- CLOCK – běží jen 2 ms → simulace neskončí timeoutem
    --------------------------------------------------------------------
    p_clk : process
    begin
        while now < 2 ms loop
            clk <= '0';
            wait for CLK_T/2;
            clk <= '1';
            wait for CLK_T/2;
        end loop;
        wait;
    end process;

    --------------------------------------------------------------------
    -- STIMULI
    --------------------------------------------------------------------
    p_stim : process
    begin
        ----------------------------------------------------------------
        -- RESET
        ----------------------------------------------------------------
        rst <= '1';
        wait for 200 ns;
        rst <= '0';
        wait for 200 ns;

        ----------------------------------------------------------------
        -- 1) Simulace zákmitu tlačítka (bounce)
        ----------------------------------------------------------------
        -- 0101010101 → debounce to musí ignorovat
        btn_in <= '1'; wait for 20 ns;
        btn_in <= '0'; wait for 20 ns;
        btn_in <= '1'; wait for 20 ns;
        btn_in <= '0'; wait for 20 ns;
        btn_in <= '1'; wait for 20 ns;

        ----------------------------------------------------------------
        -- 2) Krátký stisk (btn_press = 1 na 1 clock)
        ----------------------------------------------------------------
        btn_in <= '1';
        wait for 1 ms;   -- delší než debounce filtr
        btn_in <= '0';
        wait for 1 ms;

        ----------------------------------------------------------------
        -- 3) Dlouhý stisk (btn_state = 1, press jen jednou)
        ----------------------------------------------------------------
        btn_in <= '1';
        wait for 1 ms;
        btn_in <= '0';
        wait for 1 ms;

        ----------------------------------------------------------------
        -- KONEC SIMULACE
        ----------------------------------------------------------------
        report "END OF SIMULATION" severity note;
        wait;
    end process;

end architecture tb;
