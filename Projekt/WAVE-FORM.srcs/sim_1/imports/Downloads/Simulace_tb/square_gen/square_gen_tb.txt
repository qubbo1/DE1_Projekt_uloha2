library ieee;
use ieee.std_logic_1164.all;
use ieee.numeric_std.all;

entity tb_square_gen is
end tb_square_gen;

architecture tb of tb_square_gen is

    component square_gen
        port (
            phase    : in  std_logic_vector(7 downto 0);
            wave_out : out std_logic_vector(7 downto 0)
        );
    end component;

    signal phase    : std_logic_vector(7 downto 0) := (others => '0');
    signal wave_out : std_logic_vector(7 downto 0);

begin

    -- DUT
    dut : square_gen
        port map (
            phase    => phase,
            wave_out => wave_out
        );

    --------------------------------------------------------------------------
    -- STIMULI
    --------------------------------------------------------------------------
    stimuli : process
        variable expected : std_logic_vector(7 downto 0);
    begin
        ----------------------------------------------------------------------
        -- TEST 1: phase = 00 -> LOW
        ----------------------------------------------------------------------
        phase <= x"00";
        wait for 10 ns;
        expected := x"00";
        assert wave_out = expected
            report "phase=00 -> expected LOW (00)" severity error;

        ----------------------------------------------------------------------
        -- TEST 2: phase = 7F -> LOW
        ----------------------------------------------------------------------
        phase <= x"7F";
        wait for 10 ns;
        expected := x"00";
        assert wave_out = expected
            report "phase=7F -> expected LOW (00)" severity error;

        ----------------------------------------------------------------------
        -- TEST 3: phase = 80 -> HIGH
        ----------------------------------------------------------------------
        phase <= x"80";
        wait for 10 ns;
        expected := x"FF";
        assert wave_out = expected
            report "phase=80 -> expected HIGH (FF)" severity error;

        ----------------------------------------------------------------------
        -- TEST 4: phase = FF -> HIGH
        ----------------------------------------------------------------------
        phase <= x"FF";
        wait for 10 ns;
        expected := x"FF";
        assert wave_out = expected
            report "phase=FF -> expected HIGH (FF)" severity error;

        ----------------------------------------------------------------------
        -- TEST 5: full sweep 0 to 255
        ----------------------------------------------------------------------
        for i in 0 to 255 loop
            phase <= std_logic_vector(to_unsigned(i, 8));
            wait for 1 ns;

            if i < 128 then
                expected := x"00";
            else
                expected := x"FF";
            end if;

            assert wave_out = expected
                report "Mismatch at phase=" & integer'image(i)
                severity error;
        end loop;

        ----------------------------------------------------------------------
        -- END SIMULATION
        ----------------------------------------------------------------------
        wait;
    end process;

end tb;
